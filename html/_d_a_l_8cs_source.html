352 public static (int porAtender, int emAtendimento, int atendido) GetTicketCounts()
353 {
354 string query = @"
355 SELECT
356 SUM(CASE WHEN status_ticket = 'porAtender' THEN 1 ELSE 0 END) AS PorAtender,
357 SUM(CASE WHEN status_ticket = 'emAtendimento' THEN 1 ELSE 0 END) AS EmAtendimento,
358 SUM(CASE WHEN status_ticket = 'atendido' THEN 1 ELSE 0 END) AS Atendido
359 FROM Ticket";
360
361 using (SqlConnection con = new SqlConnection(conStr))
362 {
363 SqlCommand cmd = new SqlCommand(query, con);
364 con.Open();
365 SqlDataReader reader = cmd.ExecuteReader();
366
367 int porAtender = 0, emAtendimento = 0, atendido = 0;
368
369 if (reader.Read())
370 {
371 porAtender = reader.GetInt32(reader.GetOrdinal("PorAtender"));
372 emAtendimento = reader.GetInt32(reader.GetOrdinal("EmAtendimento"));
373 atendido = reader.GetInt32(reader.GetOrdinal("Atendido"));
374 }
375
376 return (porAtender, emAtendimento, atendido);
377 }
