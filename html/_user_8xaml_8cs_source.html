141 private void btnSubmit_Click(object sender, RoutedEventArgs e)
142 {
143 // Obter o ID do colaborador atual
144 string colaboradorId = User.getID();
145
146 // Determinar o tipo de problema
147 string tipo = rdbHard.IsChecked == true ? "hardware" : "software";
148
149 // Coletar a descrição da tarefa e a descrição do problema
150 string task = txtTask.Text;
151 string desc = txtDesc.Text;
152
153 // Obter a data atual
154 string dataCriacao = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss");
155
156 // SQL para inserir um novo ticket
157 string sql = $"INSERT INTO Ticket (Colaborador_ID_Maker, tipo, data_criacao, status_ticket, task_ticket, info_ticket) " +
158 $"VALUES ({colaboradorId}, '{tipo}', '{dataCriacao}', 'porAtender', '{task}', '{desc}')";
159
160 // Executar o comando SQL usando a camada DAL
161 bool result = DAL.ExecBD(sql);
162
163 // Verificar se a inserção foi bem-sucedida
164 if (result)
165 {
166 MessageBox.Show("Ticket inserido com sucesso!", "Sucesso", MessageBoxButton.OK, MessageBoxImage.Information);
167 }
168 else
169 {
170 MessageBox.Show("Erro ao inserir o ticket.", "Erro", MessageBoxButton.OK, MessageBoxImage.Error);
171 }
172 }
